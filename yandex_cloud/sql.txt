# Создание таблиц

CREATE TABLE IF NOT EXISTS questions (
    id Uint64,
    question String,
    options Utf8,
    correct_option Uint64,
    PRIMARY KEY (id)
);


CREATE TABLE IF NOT EXISTS user_stats (
    user_id Uint64,
    correct_answers Uint64,
    total_questions Uint64,
    PRIMARY KEY (user_id)
);

COMMIT ;
